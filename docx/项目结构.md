**user.service.ts 示例**:
```typescript
import prisma from '../../config/database'
import { wechatService } from '../../external/wechat'
import { jwtService } from '../../shared/utils/jwt'

export class UserService {
  // 微信登录
  async login(code: string) {
    // 1. 通过code换取openid
    const { openid, session_key } = await wechatService.code2Session(code)

    // 2. 查询或创建用户
    let user = await prisma.user.findUnique({ where: { openid } })
    if (!user) {
      user = await prisma.user.create({
        data: { openid, sessionKey: session_key }
      })
    }

    // 3. 生成JWT Token
    const token = jwtService.generate({ userId: user.id })

    // 4. 检查是否设置过偏好
    const preference = await prisma.userPreference.findUnique({
      where: { userId: user.id }
    })

    return {
      token,
      userInfo: this.formatUser(user),
      hasPreference: !!preference
    }
  }

  // 获取用户信息
  async getProfile(userId: string) {
    const user = await prisma.user.findUnique({
      where: { id: userId },
      include: {
        _count: {
          select: { checks: true, comments: true, routes: true }
        }
      }
    })

    return {
      ...this.formatUser(user),
      stats: {
        checkCount: user._count.checks,
        commentCount: user._count.comments,
        routeCount: user._count.routes
      }
    }
  }

  // 更新用户信息
  async updateProfile(userId: string, data: UpdateProfileDto) {
    const user = await prisma.user.update({
      where: { id: userId },
      data
    })
    return this.formatUser(user)
  }

  // 提交偏好
  async setPreference(userId: string, data: PreferenceDto) {
    await prisma.userPreference.upsert({
      where: { userId },
      update: data,
      create: { userId, ...data }
    })
  }

  // 格式化用户数据
  private formatUser(user: User) {
    return {
      id: user.id,
      openid: user.openid,
      nickname: user.nickname,
      avatar: user.avatar,
      phone: user.phone ? this.maskPhone(user.phone) : undefined,
      gender: user.gender,
      locale: user.locale
    }
  }

  // 手机号脱敏
  private maskPhone(phone: string) {
    return phone.replace(/(\d{3})\d{4}(\d{4})/, '$1****$2')
  }
}

export const userService = new UserService()
```

**auth.ts 示例**:
```typescript
import { Context, Next } from 'koa'
import { jwtService } from '../utils/jwt'
import { ErrorCode } from '../utils/response'

export const authMiddleware = async (ctx: Context, next: Next) => {
  const authHeader = ctx.headers.authorization

  if (!authHeader || !authHeader.startsWith('Bearer ')) {
    ctx.status = 401
    ctx.body = { code: ErrorCode.UNAUTHORIZED, message: '请先登录', data: null }
    return
  }

  const token = authHeader.slice(7)

  try {
    const payload = jwtService.verify(token)
    ctx.state.user = payload
    await next()
  } catch (err) {
    ctx.status = 401
    ctx.body = { code: ErrorCode.TOKEN_INVALID, message: '登录已过期', data: null }
  }
}
```

```typescript
import { Context } from 'koa'

export enum ErrorCode {
  SUCCESS = 0,
  PARAM_ERROR = 40001,
  MISSING_PARAM = 40002,
  FORMAT_ERROR = 40003,
  UNAUTHORIZED = 40101,
  TOKEN_INVALID = 40102,
  FORBIDDEN = 40301,
  NOT_FOUND = 40401,
  DATA_NOT_FOUND = 40402,
  SERVER_ERROR = 50001,
  SERVICE_UNAVAILABLE = 50002,
  AI_SERVICE_ERROR = 50003
}

export interface ApiResponse<T = any> {
  code: number
  message: string
  data: T
}

export function successResponse<T>(ctx: Context, data: T, message = 'success') {
  ctx.body = {
    code: ErrorCode.SUCCESS,
    message,
    data
  } as ApiResponse<T>
}

export function errorResponse(ctx: Context, code: ErrorCode, message: string) {
  ctx.status = code >= 40001 && code < 50000 ? 400 : code >= 50000 ? 500 : 200
  ctx.body = {
    code,
    message,
    data: null
  } as ApiResponse
}

export class BusinessError extends Error {
  constructor(public code: ErrorCode, message: string) {
    super(message)
    this.name = 'BusinessError'
  }
}
```

**router.ts 示例**:
```typescript
import { openaiService } from './openai'
import prisma from '../config/database'
import { ROUTE_GENERATE_PROMPT } from './prompt'

export class AIRouteService {
  // 生成路线
  async generateRoute(params: {
    prompt: string
    days: number
    userId: string
    preferences?: string[]
  }) {
    // 1. 解析用户需求
    const parsed = await this.parseUserInput(params.prompt)

    // 2. 获取候选POI
    const candidatePois = await this.getCandidatePois(parsed, params.preferences)

    // 3. 调用LLM生成路线
    const routeData = await openaiService.generateRoute({
      userInput: params.prompt,
      days: params.days,
      pois: candidatePois
    })

    // 4. 保存路线
    const route = await prisma.route.create({
      data: {
        userId: params.userId,
        routeName: { zh: routeData.title, en: routeData.title },
        totalDays: params.days,
        description: { zh: params.prompt, en: params.prompt },
        isAiGenerated: true,
        sourcePrompt: params.prompt,
        tags: parsed.tags,
        pois: {
          create: routeData.schedule.flatMap(day =>
            day.pois.map((poi, idx) => ({
              poiId: poi.id,
              dayNumber: day.day,
              visitOrder: idx + 1,
              stayDuration: poi.stayDuration,
              visitNote: { zh: poi.reason, en: poi.reason }
            }))
          )
        }
      }
    })

    return this.formatRouteResponse(route)
  }

  // 解析用户输入
  private async parseUserInput(prompt: string) {
    const response = await openaiService.chat([
      { role: 'user', content: `从以下用户需求中提取关键信息：${prompt}` }
    ], `提取天数、偏好关键词`)
    return JSON.parse(response)
  }

  // 获取候选POI
  private async getCandidatePois(parsed: ParsedInput, tags?: string[]) {
    const searchTags = tags || parsed.tags || []

    const pois = await prisma.poiInfo.findMany({
      where: {
        status: 1,
        poiTags: {
          some: {
            tag: { tagCode: { in: searchTags } },
            weight: { gte: 0.5 }
          }
        }
      },
      include: {
        poiTags: { include: { tag: true } }
      },
      orderBy: { stats: { heatScore: 'desc' } },
      take: 50
    })

    return pois.map(p => ({
      id: p.id,
      name: p.poiName,
      description: p.description,
      latitude: p.latitude.toNumber(),
      longitude: p.longitude.toNumber(),
      tags: p.poiTags.map(t => t.tag.tagCode)
    }))
  }

  // 格式化路线响应
  private async formatRouteResponse(route: any) {
    const fullRoute = await prisma.route.findUnique({
      where: { id: route.id },
      include: {
        pois: {
          include: { poi: true },
          orderBy: [{ dayNumber: 'asc' }, { visitOrder: 'asc' }]
        }
      }
    })

    // 转换为前端期望的格式
    return {
      id: fullRoute.id,
      title: fullRoute.routeName,
      days: fullRoute.totalDays,
      totalPoi: fullRoute.pois.length,
      schedule: this.groupByDay(fullRoute.pois),
      tags: fullRoute.tags,
      createdAt: fullRoute.createdAt
    }
  }
}

export const aiRouteService = new AIRouteService()
```

**amap.ts 示例**:
```typescript
import axios from 'axios'
import crypto from 'crypto'
import { config } from '../config'

interface AMapPOI {
  id: string
  name: string
  location: string
  address: string
  type: string
}

export class AMapService {
  private key: string
  private secret?: string
  private baseUrl = 'https://restapi.amap.com/v3'

  constructor() {
    this.key = config.amap.key
    this.secret = config.amap.secret
  }

  // 搜索附近POI
  async searchNearby(params: {
    location: [number, number]
    keywords?: string
    types?: string
    radius?: number
  }) {
    const url = `${this.baseUrl}/place/around`
    const response = await axios.get(url, {
      params: {
        key: this.key,
        location: params.location.join(','),
        keywords: params.keywords,
        types: params.types,
        radius: params.radius || 5000,
        output: 'json'
      }
    })

    return response.data.pois as AMapPOI[]
  }

  // 地理编码
  async geocode(address: string, city?: string) {
    const url = `${this.baseUrl}/geocode/geo`
    const response = await axios.get(url, {
      params: { key: this.key, address, city, output: 'json' }
    })

    const geocodes = response.data.geocodes
    if (geocodes && geocodes.length > 0) {
      return geocodes[0]
    }
    return null
  }
}
```

## 五、应用入口

### 5.1 app.ts

```typescript
import Koa from 'koa'
import bodyParser from 'koa-bodyparser'
import cors from 'koa-cors'
import { publicRoutes, privateRoutes } from './routes'
import { authMiddleware } from './shared/middleware/auth'
import { errorMiddleware } from './shared/middleware/error'
import { loggerMiddleware } from './shared/middleware/logger'
import { i18nMiddleware } from './shared/middleware/i18n'

export function createApp() {
  const app = new Koa()

  // 全局中间件
  app.use(errorMiddleware)        // 错误处理
  app.use(loggerMiddleware)        // 日志记录
  app.use(cors())                 // 跨域
  app.use(bodyParser())           // 请求体解析
  app.use(i18nMiddleware)         // 国际化

  // 公开路由
  app.use(publicRoutes.routes())
  app.use(publicRoutes.allowedMethods())

  // 认证中间件
  app.use(async (ctx, next) => {
    if (ctx.path.startsWith('/api/')) {
      // 不需要认证的路径
      const publicPaths = ['/api/user/login']
      if (publicPaths.includes(ctx.path)) {
        await next()
      } else {
        await authMiddleware(ctx, next)
      }
    } else {
      await next()
    }
  })

  // 私有路由
  app.use(privateRoutes.routes())
  app.use(privateRoutes.allowedMethods())

  return app
}
```
